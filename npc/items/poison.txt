// Description:
//    Poison effect of some items with parameters:
//      @r   randomness of bad luck to be poisoned,
//      @h   heal value,
//      @p   poison value (not implemented yet).

function	script	usePoison	{
	set @temp, rand(@r);
	if (@temp == 0) sc_start sc_poison, 1, 50;
	set @temp, 0;
	set @r, 0;
	itemheal @h, 0;
	set @h, 0;
	return;
}
