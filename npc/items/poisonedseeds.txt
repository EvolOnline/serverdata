// Description:
//    3/4 chances to get poisoned.
//    1/4 chances to get healed (1 HP).

function	script	usePoisonedSeeds	{
	set @temp, rand(4);
	if (!@temp) itemheal 1, 0;
	if (@temp) sc_start sc_poison, 1, 50;
	set @temp, 0;
	return;
}
