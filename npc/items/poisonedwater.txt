// Description:
//    1/4 chances to get poisoned.
//    3/4 chances to get healed (15 HP).

function	script	usePoisonedWater	{
	set @temp, rand(4);
	if (!@temp) sc_start sc_poison, 1, 50;
	if (@temp) itemheal 15, 0;
	set @temp, 0;
	return;
}
