000-2-1.gat,61,34,0,1	script	AreaNPC	0,0,1,{

OnTouch:
    set @q_julia, geta2(ShipQuests, ShipQuests_Julia);
    if (@q_julia == 0) goto l_TalkToJulia;
    if (BaseLevel < 5) goto l_TooWeak;
    warp "000-2-2.gat", 48, 28;

    close;

L_TooWeak:
    doevent "Peter::OnTooWeak";

    close;

l_TalkToJulia:
    doevent "Peter::OnJulia";

    close;

}

000-2-1.gat,60,35,0,1	script	Peter	303,{

    set @q_julia, geta2(ShipQuests, ShipQuests_Julia);
    if (@q_julia == 0) goto L_TalkToJulia;
    if (BaseLevel < 5) goto OnTooWeak;

    mesn;
    mes l("Did you see... This thing?");
    next;

    menu
        l("What are you talking about?"), -,
        l("Yes I did."), l_Sawit;

    mes "";
    mesn;
    mes l("There's something which exceed any imagination behind this door... It's not human.");
    next;

    menu
        l("I don't understand."), -,
        l("It's just an orange ball, nothing else."), l_Sawit;

    mes "";
    mesn;
    mes l("Did you see all of these pumkpins monsters around here?");
    next;
    mes l("Well, there's something like... Their king, behind this door.");
    next;
    mes l("Be prudent if you go in there, it could be bad that we saved you from the ocean for nothing!");

    close;

l_Sawit:
    mes "";
    mesn;
    mesq l("I can't close my eyes the night, it's haunting me...");

    close;

OnJulia:
    mesn;
    mesq l("Hey, you should go see Julia to be registered on the ship board.");
    goto L_Close;

OnTooWeak:
    mesn;
    mesq lg("I won't let you enter, you're way too weak.");
    goto L_Close;

L_Close:
    set @q, 0;
    set @q_julia, 0;

    close;

}
